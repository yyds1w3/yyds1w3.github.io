const int MAXN = 2e5 + 1;
int head[MAXN], pos[MAXN], nxt[MAXN], tot;
int c[MAXN];
int n, k;
void init() {
    for (int i = 1; i <= n; ++i) {head[i] = -1; c[i] = 0;}
    tot = 0;
}
void insert(int val, int p) {
    ++tot;
    pos[tot] = p;
    nxt[tot] = head[val];
    head[val] = tot;
}
void solve() {
    cin >> n >> k;
    init();
    int mx_val = 0;
    for (int i = 1; i <= n; ++i) {
        int v; cin >> v;
        insert(v, i);
        mx_val = max(v, mx_val);
    }
    vector<int> paint;
    for (int i = mx_val; i >= 1; --i) {
        if (head[i] == -1) continue;
        int count = 0;
        for (int j = head[i]; j != -1; j = nxt[j]) {
            if (count < k) {
                paint.push_back(pos[j]);
                count++;
            }
        }
    }
    int painted = paint.size() / k * k;
    for (int i = 0; i < painted; ++i) c[paint[i]] = i % k + 1;
    for (int i = 1; i <= n; ++i) cout << c[i] << " ";
    cout << "\n";
}
